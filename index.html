        function onSkillChange(slotId, skillNum) {

            const select = document.querySelector(`.officer-slot[data-slot="${slotId}"] .skill-item:nth-child(${skillNum + 1}) select`);

            const skillId = parseInt(select.value);

            const skill = skillsData.find(s => s.id === skillId);

            if (skill) {

                console.log(`[戰法選擇] 武將槽 ${slotId} 戰法${skillNum}: ${skill.name}`);

            }

        }


        function updateTotalCost() {

            let totalCost = 0;

            document.querySelectorAll('.officer-select').forEach(select => {

                const id = parseInt(select.value);

                if (id) {

                    const officer = officersData.find(o => o.id === id);

                    totalCost += officer.cost;

                }

            });

            

            const costElement = document.getElementById('totalCost');

            costElement.textContent = `COST: ${totalCost} / 20`;

            costElement.className = "cost-display " + 

                (totalCost > 20 ? "cost-danger" : 

                 totalCost >= 18 ? "cost-warning" : "cost-normal");

        }


        function updatePreviewStats() {

            let leadSum = 0, strSum = 0, intSum = 0, spdSum = 0;

            document.querySelectorAll('.officer-select').forEach(select => {

                const id = parseInt(select.value);

                if (id) {

                    const officer = officersData.find(o => o.id === id);

                    leadSum += officer.stats.lead;

                    strSum += officer.stats.str;

                    intSum += officer.stats.int;

                    spdSum += officer.stats.spd;

                }

            });

            document.getElementById('sumLeadership').textContent = leadSum;

            document.getElementById('sumStrength').textContent = strSum;

            document.getElementById('sumIntelligence').textContent = intSum;

            document.getElementById('sumSpeed').textContent = spdSum;

        }
